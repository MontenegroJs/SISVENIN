@classmethod
def _get_connection(cls) -> sqlite3.Connection:
    """Obtiene conexión a la base de datos (método interno)"""
    os.makedirs(os.path.dirname(cls.DB_PATH), exist_ok=True)
    return sqlite3.connect(cls.DB_PATH)

Clase base (referencia)
python
class ProductoRepositorio:
    DB_PATH = "..."
    
    @classmethod
    def _get_connection(cls) -> sqlite3.Connection:
        ...
    
    @classmethod
    def inicializar(cls):
        """Inicializa la BD (crea tablas si no existen)"""
        cls.crear_tabla()
    
    @classmethod
    def crear_tabla(cls):
        """Crea la tabla si no existe"""
        conn = cls._get_connection()
        cursor = conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS productos (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                nombre TEXT NOT NULL,
                precio REAL NOT NULL,
                stock INTEGER NOT NULL,
                activo INTEGER DEFAULT 1
            )
        """)
        conn.commit()
        conn.close()
    
    @classmethod
    def guardar(cls, obj: ProductoModelo) -> ProductoModelo:
        """Guarda un nuevo producto (INSERT)"""
        conn = cls._get_connection()
        cursor = conn.cursor()
        cursor.execute(
            "INSERT INTO productos (nombre, precio, stock, activo) VALUES (?, ?, ?, ?)",
            (obj.nombre, obj.precio, obj.stock, 1 if obj.activo else 0)
        )
        conn.commit()
        obj.id = cursor.lastrowid
        conn.close()
        return obj
    
    @classmethod
    def obtener_todos(cls) -> List[ProductoModelo]:
        """Obtiene todos los productos activos"""
        conn = cls._get_connection()
        cursor = conn.cursor()
        cursor.execute("SELECT id, nombre, precio, stock, activo FROM productos WHERE activo = 1")
        rows = cursor.fetchall()
        conn.close()
        
        productos = []
        for row in rows:
            obj = ProductoModelo(id=row[0], nombre=row[1], precio=row[2], stock=row[3])
            obj.activo = bool(row[4])
            productos.append(obj)
        return productos
    
    @classmethod
    def obtener_por_id(cls, obj_id: int) -> Optional[ProductoModelo]:
        """Obtiene un producto por su ID"""
        conn = cls._get_connection()
        cursor = conn.cursor()
        cursor.execute("SELECT id, nombre, precio, stock, activo FROM productos WHERE id = ? AND activo = 1", (obj_id,))
        row = cursor.fetchone()
        conn.close()
        
        if row:
            obj = ProductoModelo(id=row[0], nombre=row[1], precio=row[2], stock=row[3])
            obj.activo = bool(row[4])
            return obj
        return None
    
    @classmethod
    def actualizar(cls, obj: ProductoModelo) -> None:
        """Actualiza un producto existente"""
        conn = cls._get_connection()
        cursor = conn.cursor()
        cursor.execute(
            "UPDATE productos SET nombre = ?, precio = ?, stock = ?, activo = ? WHERE id = ?",
            (obj.nombre, obj.precio, obj.stock, 1 if obj.activo else 0, obj.id)
        )
        conn.commit()
        conn.close()
    
    @classmethod
    def eliminar(cls, obj_id: int) -> None:
        """Eliminado lógico (activo = 0)"""
        conn = cls._get_connection()
        cursor = conn.cursor()
        cursor.execute("UPDATE productos SET activo = 0 WHERE id = ?", (obj_id,))
        conn.commit()
        conn.close()

def test_guardar_y_recuperar(repo_producto_temp):
    ProductoRepositorio = repo_producto_temp
    from src.app.modules.producto.producto_modelo import ProductoModelo

Reglas de integridad (a nivel de repositorio)
Regla	Dónde se valida	Método
Precio > 0	Controlador	validar_precio()
Stock >= 0	Controlador	validar_stock()
Nombre no vacío	Controlador	validar_nombre()
ID existe antes de actualizar	Repositorio	obtener_por_id()
Buenas prácticas
Usar borrado lógico (activo = 0) en lugar de DELETE

Siempre hacer commit después de INSERT/UPDATE/DELETE

Cerrar conexiones en finally o automáticamente con context managers

No mezclar reglas de negocio en el repositorio (solo acceso a datos)

Nombres de tablas en plural (productos, clientes)
