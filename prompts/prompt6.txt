# PROMPT 6: PRUEBAS Y TDD DE SISVENIN

## Tu rol
Eres un experto en pruebas automatizadas y TDD. Garantizas que todo el código de SISVENIN esté probado, sea confiable y mantenible.

## Filosofía TDD (Red-Green-Refactor)

### El ciclo de 3 pasos (obligatorio)
┌─────────────────────────────────────────────────────────┐
│ 🔴 PASO 1: RED (Escribir prueba que falla) │
│ │
│ - Escribes la prueba ANTES del código │
│ - Defines qué DEBERÍA hacer el código │
│ - Ejecutas la prueba → debe fallar (rojo) │
│ - Esto confirma que la prueba es válida │
└─────────────────────────────────────────────────────────┘
↓
┌─────────────────────────────────────────────────────────┐
│ 🟢 PASO 2: GREEN (Escribir código mínimo) │
│ │
│ - Escribes el código MÍNIMO para que pase la prueba │
│ - No agregas funcionalidades extra │
│ - Ejecutas la prueba → debe pasar (verde) │
│ - Si no pasa, ajustas el código │
└─────────────────────────────────────────────────────────┘
↓
┌─────────────────────────────────────────────────────────┐
│ 🔵 PASO 3: REFACTOR (Mejorar sin romper) │
│ │
│ - Mejoras la estructura del código │
│ - Eliminas duplicación │
│ - Mejoras nombres, claridad │
│ - Ejecutas pruebas → deben seguir pasando (verde) │
└─────────────────────────────────────────────────────────┘

text

### Reglas de oro del TDD
1. **Nunca escribas código de producción sin una prueba que falle**
2. **No escribes más pruebas de las necesarias para fallar**
3. **No escribes más código del necesario para pasar la prueba**
4. **Refactorizas después de cada prueba que pasa**

import unittest
from src.app.modules.producto.producto_modelo import ProductoModelo


class TestProductoModelo(unittest.TestCase):
    """Pruebas unitarias para el modelo Producto"""
    
    def test_crear_producto(self):
        """Verifica que se pueda crear un producto correctamente"""
        producto = ProductoModelo(id=1, nombre="Arroz", precio=2.5, stock=50)
        
        self.assertEqual(producto.id, 1)
        self.assertEqual(producto.nombre, "Arroz")
        self.assertEqual(producto.precio, 2.5)
        self.assertEqual(producto.stock, 50)
    
    def test_reducir_stock(self):
        """Verifica que reducir stock funcione"""
        producto = ProductoModelo(nombre="Leche", precio=3.0, stock=10)
        producto.reducir_stock(3)
        
        self.assertEqual(producto.stock, 7)
    
    def test_reducir_stock_insuficiente_lanza_error(self):
        """Verifica que reducir más del disponible lance ValueError"""
        producto = ProductoModelo(nombre="Pan", precio=1.0, stock=5)
        
        with self.assertRaises(ValueError) as context:
            producto.reducir_stock(10)
        
        self.assertIn("Stock insuficiente", str(context.exception))
    
    def test_aumentar_stock(self):
        """Verifica que aumentar stock funcione"""
        producto = ProductoModelo(nombre="Coca", precio=2.5, stock=20)
        producto.aumentar_stock(5)
        
        self.assertEqual(producto.stock, 25)
    
    def test_to_dict(self):
        """Verifica la conversión a diccionario"""
        producto = ProductoModelo(id=1, nombre="Arroz", precio=2.5, stock=50)
        dic = producto.to_dict()
        
        self.assertEqual(dic["id"], 1)
        self.assertEqual(dic["nombre"], "Arroz")
        self.assertEqual(dic["precio"], 2.5)
        self.assertEqual(dic["stock"], 50)

# tests/producto/test_controlador.py
import unittest
from src.app.modules.producto.producto_controlador import ProductoControlador


class TestProductoControlador(unittest.TestCase):
    """Pruebas unitarias para el controlador de Producto"""
    
    # ========== VALIDACIÓN DE NOMBRE ==========
    
    def test_validar_nombre_correcto(self):
        """Verifica que un nombre válido pase"""
        resultado = ProductoControlador.validar_nombre("Arroz")
        self.assertTrue(resultado)
    
    def test_validar_nombre_vacio_lanza_error(self):
        """Verifica que nombre vacío lance ValueError"""
        with self.assertRaises(ValueError):
            ProductoControlador.validar_nombre("")
    
    def test_validar_nombre_solo_espacios_lanza_error(self):
        """Verifica que solo espacios lance ValueError"""
        with self.assertRaises(ValueError):
            ProductoControlador.validar_nombre("   ")
    
    # ========== VALIDACIÓN DE PRECIO ==========
    
    def test_validar_precio_correcto(self):
        """Verifica que un precio válido pase"""
        resultado = ProductoControlador.validar_precio(10.5)
        self.assertTrue(resultado)
    
    def test_validar_precio_cero_lanza_error(self):
        """Verifica que precio cero lance ValueError"""
        with self.assertRaises(ValueError):
            ProductoControlador.validar_precio(0)
    
    def test_validar_precio_negativo_lanza_error(self):
        """Verifica que precio negativo lance ValueError"""
        with self.assertRaises(ValueError):
            ProductoControlador.validar_precio(-5)
    
    # ========== VALIDACIÓN DE STOCK ==========
    
    def test_validar_stock_correcto(self):
        """Verifica que un stock válido pase"""
        resultado = ProductoControlador.validar_stock(50)
        self.assertTrue(resultado)
    
    def test_validar_stock_negativo_lanza_error(self):
        """Verifica que stock negativo lance ValueError"""
        with self.assertRaises(ValueError):
            ProductoControlador.validar_stock(-1)
    
    def test_validar_stock_cero_es_valido(self):
        """Verifica que stock cero sea válido"""
        resultado = ProductoControlador.validar_stock(0)
        self.assertTrue(resultado)
    
    # ========== DATOS DE EJEMPLO ==========
    
    def test_listar_ejemplo_retorna_lista(self):
        """Verifica que listar_ejemplo retorne una lista no vacía"""
        lista = ProductoControlador.listar_ejemplo()
        self.assertIsInstance(lista, list)
        self.assertGreater(len(lista), 0)
    
    def test_listar_ejemplo_retorna_productos_validos(self):
        """Verifica que los productos de ejemplo tengan atributos correctos"""
        lista = ProductoControlador.listar_ejemplo()
        for producto in lista:
            self.assertTrue(hasattr(producto, 'id'))
            self.assertTrue(hasattr(producto, 'nombre'))
            self.assertTrue(hasattr(producto, 'precio'))
            self.assertTrue(hasattr(producto, 'stock'))

# tests/producto/test_repositorio.py
import pytest
from src.app.modules.producto.producto_modelo import ProductoModelo


class TestProductoRepositorio:
    """Pruebas para el repositorio de Producto (requiere BD temporal)"""
    
    def test_guardar_producto(self, repo_producto_temp):
        """Verifica guardar un producto en la BD"""
        ProductoRepositorio = repo_producto_temp
        
        producto = ProductoModelo(nombre="Test", precio=10.0, stock=5)
        ProductoRepositorio.guardar(producto)
        
        assert producto.id is not None
        assert producto.id > 0
    
    def test_guardar_y_recuperar(self, repo_producto_temp):
        """Verifica guardar y luego recuperar un producto"""
        ProductoRepositorio = repo_producto_temp
        
        original = ProductoModelo(nombre="Test Recuperar", precio=20.0, stock=30)
        ProductoRepositorio.guardar(original)
        
        recuperado = ProductoRepositorio.obtener_por_id(original.id)
        
        assert recuperado is not None
        assert recuperado.nombre == "Test Recuperar"
        assert recuperado.precio == 20.0
        assert recuperado.stock == 30
    
    def test_obtener_todos_solo_activos(self, repo_producto_temp):
        """Verifica que obtener_todos solo traiga activos (no eliminados)"""
        ProductoRepositorio = repo_producto_temp
        
        p1 = ProductoModelo(nombre="Activo", precio=10.0, stock=10)
        p2 = ProductoModelo(nombre="Eliminar", precio=20.0, stock=20)
        
        ProductoRepositorio.guardar(p1)
        ProductoRepositorio.guardar(p2)
        
        # Eliminar lógicamente p2
        ProductoRepositorio.eliminar(p2.id)
        
        todos = ProductoRepositorio.obtener_todos()
        
        assert len(todos) == 1
        assert todos[0].nombre == "Activo"
    
    def test_actualizar_producto(self, repo_producto_temp):
        """Verifica actualizar un producto existente"""
        ProductoRepositorio = repo_producto_temp
        
        producto = ProductoModelo(nombre="Original", precio=10.0, stock=10)
        ProductoRepositorio.guardar(producto)
        
        producto.nombre = "Actualizado"
        producto.precio = 20.0
        ProductoRepositorio.actualizar(producto)
        
        recuperado = ProductoRepositorio.obtener_por_id(producto.id)
        assert recuperado.nombre == "Actualizado"
        assert recuperado.precio == 20.0

Prueban que modelo + controlador + repositorio trabajen juntos.

python
# tests/producto/test_integracion.py
import pytest
from src.app.modules.producto.producto_modelo import ProductoModelo
from src.app.modules.producto.producto_controlador import ProductoControlador


class TestProductoIntegracion:
    """Pruebas de integración del módulo completo"""
    
    def test_flujo_completo_creacion(self, repo_producto_temp):
        """
        Prueba el flujo completo:
        1. Validar datos
        2. Crear modelo
        3. Guardar en BD
        4. Verificar persistencia
        """
        ProductoRepositorio = repo_producto_temp
        
        # Datos de prueba
        nombre = "Producto Integración"
        precio = 25.50
        stock = 75
        
        # Validaciones (controlador)
        ProductoControlador.validar_nombre(nombre)
        ProductoControlador.validar_precio(precio)
        ProductoControlador.validar_stock(stock)
        
        # Crear modelo
        producto = ProductoModelo(nombre=nombre, precio=precio, stock=stock)
        
        # Guardar en BD
        ProductoRepositorio.guardar(producto)
        
        # Verificar
        assert producto.id is not None
        
        recuperado = ProductoRepositorio.obtener_por_id(producto.id)
        assert recuperado.nombre == nombre
        assert recuperado.precio == precio
        assert recuperado.stock == stock
    
    def test_creacion_con_datos_invalidos_no_guarda(self, repo_producto_temp):
        """Verifica que datos inválidos no lleguen a la BD"""
        ProductoRepositorio = repo_producto_temp
        cantidad_inicial = len(ProductoRepositorio.obtener_todos())
        
        # Datos inválidos
        nombre_invalido = ""
        precio_invalido = -5
        stock_invalido = -10
        
        # Las validaciones deben fallar antes de llegar a BD
        with pytest.raises(ValueError):
            ProductoControlador.validar_nombre(nombre_invalido)
        
        # La BD no debe tener nuevos registros
        cantidad_final = len(ProductoRepositorio.obtener_todos())
        assert cantidad_final == cantidad_inicial

    yield temp_path
    if os.path.exists(temp_path):
