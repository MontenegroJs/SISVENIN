        header = self.horizontalHeader()

        header.setSectionResizeMode(1, QHeaderView.Stretch)           # Nombre
        header.setSectionResizeMode(2, QHeaderView.ResizeToContents)  # Precio

def cargar_productos(self, productos: List[ProductoModelo]):
    """Carga una lista de productos en la tabla"""
    self.setRowCount(len(productos))
    
    for i, producto in enumerate(productos):
        self.setItem(i, 0, QTableWidgetItem(str(producto.id)))
        self.setItem(i, 1, QTableWidgetItem(producto.nombre))
        self.setItem(i, 2, QTableWidgetItem(f"S/. {producto.precio:.2f}"))
        self.setItem(i, 3, QTableWidgetItem(str(producto.stock)))
        
        # Resaltar stock bajo
        if producto.stock < 10:
            item = self.item(i, 3)
            item.setBackground(QtGui.QColor(255, 200, 200))  # Rojo claro

class ProductoVista(QWidget):
    def __init__(self):
        super().__init__()
        self._setup_ui()
        self._connect_signals()
        self.refrescar()
    
    def _setup_ui(self):
        """Configura todos los widgets y layouts"""
        layout = QVBoxLayout(self)
        
        # Título
        titulo = QLabel("📦 Gestión de Productos")
        titulo.setStyleSheet("font-size: 20px; font-weight: bold;")
        layout.addWidget(titulo)
        
        # Tabla
        self.tabla = TablaProductos()
        layout.addWidget(self.tabla)
        
        # Botones
        botones_layout = QHBoxLayout()
        
        self.btn_refrescar = BotonPrimario("🔄 Refrescar")
        self.btn_nuevo = BotonPrimario("➕ Nuevo")
        self.btn_editar = BotonPrimario("✏️ Editar")
        self.btn_eliminar = BotonPeligro("🗑️ Eliminar")
        
        botones_layout.addWidget(self.btn_refrescar)
        botones_layout.addWidget(self.btn_nuevo)
        botones_layout.addWidget(self.btn_editar)
        botones_layout.addWidget(self.btn_eliminar)
        botones_layout.addStretch()
        
        layout.addLayout(botones_layout)

from PySide6.QtCore import Signal, QObject

class ContadorPersonalizado(QWidget):
    valor_cambiado = Signal(int)  # Señal personalizada
    
    def __init__(self):
        super().__init__()
        self._valor = 0
        self._setup_ui()
    
    def _setup_ui(self):
        layout = QHBoxLayout(self)
        self.btn_menos = QPushButton("-")
        self.lbl_valor = QLabel("0")
        self.btn_mas = QPushButton("+")
        
        self.btn_menos.clicked.connect(self._decrementar)
        self.btn_mas.clicked.connect(self._incrementar)
        
        layout.addWidget(self.btn_menos)
        layout.addWidget(self.lbl_valor)
        layout.addWidget(self.btn_mas)
    
    def _incrementar(self):
        self.valor += 1
    
    def _decrementar(self):
        self.valor -= 1
    
    @property
    def valor(self):
        return self._valor
    
    @valor.setter
    def valor(self, nuevo):
        self._valor = nuevo
        self.lbl_valor.setText(str(nuevo))
        self.valor_cambiado.emit(nuevo)
